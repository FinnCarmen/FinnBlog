#include<iostream>
#include<cmath>
using namespace std;
const int N = 110;
int s[N][N];
int n, m, ans = 1e9;
void solve()
{
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
        {
            int x;
            cin >> x;
            s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + x;
        }
    for(int i = 1; i < n; i++)
    {
        int x = abs(s[n][m] - 2 * s[i][m]);
        ans = min(ans, x);
    }
    for(int i = 1; i < m; i++)
    {
        int x = abs(s[n][m] - 2 * s[n][i]);
        ans = min(ans, x);
    }
    cout << ans;
}
int main()
{
    cin >> n >> m;
    solve();
    return 0;
}

    while (cin >> a >> b) {
        int sum = 0;
        // 累加区间 a 到 b 的和

class Solution {
public:
    vector<vector<int>> generateMatrix(int n) {
        vector<vector<int>>res(n,vector<int>(n,0));//*
        int x=0,y=0;
        int loop=n/2;
        int mid=n/2;
        int val=1;
        int offset=1;
        int i,j;
        while(loop--)
        {
            i=x;
            j=y;
            for(j;j<n-offset;j++)
            {
                res[i][j]=val++;
            }
            for(i;i<n-offset;i++)
            {
                res[i][j]=val++;
            }
            for(;j>y;j--)
            {
                res[i][j]=val++;
            }
            for(;i>x;i--)
            {
                res[i][j]=val++;
            }
            x++;
            y++;
            offset+=1;
        }
        if(n%2)
        {
            res[mid][mid]=val;
        }
        return res;
    }
};

利用边界旋转方向

#include <vector>
using namespace std;

class Solution {
public:
    vector<vector<int>> generateMatrix(int n) {
        int maxVal = n * n;
        int cur = 1;
        vector<vector<int>> res(n, vector<int>(n, 0)); // 初始化矩阵为0
        int row = 0, col = 0;
        // 方向数组：右, 下, 左, 上
        vector<vector<int>> dir = { {0,1}, {1,0}, {0,-1}, {-1,0} };
        int dirId = 0; // 当前方向的索引

        while(cur <= maxVal) {
            res[row][col] = cur++;
            // 计算下一个位置
            int nextRow = row + dir[dirId][0];
            int nextCol = col + dir[dirId][1];
            // 检查下一个位置是否越界或已被填充
            if(nextRow < 0 || nextRow >= n || nextCol < 0 || nextCol >= n || res[nextRow][nextCol] != 0) {
                dirId = (dirId + 1) % 4; // 顺时针切换到下一个方向
                // 更新方向后的新位置
                row += dir[dirId][0];
                col += dir[dirId][1];
            }
            else {
                row = nextRow;
                col = nextCol;
            }
        }

        return res;
    }
};

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums)
    {
        int len=0;
        int res=100005;
        for(int i=0;i<nums.size();i++)
        {
            int sum=0;
            for(int j=i;j<nums.size();j++)
            {
                sum+=nums[j];
                if(sum>=target)
                {
                    len=j-i+1;
                    res=res<len?res:len;
                    break;
                }
            }
        }
        return res==100005?0:res;
    }
};

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums)
    {
        int res = 100005;
        int sum =0;
        int i=0;
        int len =0;
        for(int j=0;j<nums.size();j++)//控制滑动窗口终点
        {
            sum+=nums[j];
            while(sum>=target)
            {
                len=j-i+1;
                res=res<len?res:len;
                sum-=nums[i++];//*控制滑动窗口起点
            }
        }
        return res==100005?0:res;
    }
};

易错点：middle 需要随着循环不断更新，故放在 while 循环内部

class Solution {
public:
    int search(vector<int>& nums, int target) {
        int l=0,r=nums.size()-1;
        while(l<=r)
        {
            int m=l+(r-l)/2;
            if(target<nums[m])r=m-1;
            else if(target>nums[m])l=m+1;
            else return m;
        }
        return -1;
    }
};

class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int j=0;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]!=val)    //*
            {
                nums[j]=nums[i];
                j++;
            }
        }
        return j;
    }
};

class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int size=nums.size();
        for(int i=0;i<size;i++)
        {
            if(nums[i]==val)    //*
            {
                for(int j=i+1;j<size;j++)
                {
                    nums[j-1]=nums[j];
                }
                i-=1;
                size-=1;
            }
        }
        return size;
    }
};
