[用栈实现队列](https://leetcode.cn/problems/implement-queue-using-stacks/)

将一个栈当作输入栈，用于压入 push 传入的数据；另一个栈当作输出栈，用于 pop 和 peek 操作。

```cpp
class MyQueue {
private:
    stack<int>in,out;
    void in2out()
    {
        while(!in.empty())
        {
            out.push(in.top());
            in.pop();
        }
    }
public:
    MyQueue() {}

    void push(int x) {
        in.push(x);
    }

    int pop() {
        if(out.empty())in2out();
        int x = out.top();
        out.pop();
        return x;
    }


    int peek() {
        if(out.empty())in2out();
        return out.top();
    }

    bool empty() {
        return in.empty() && out.empty();
    }
};
```
